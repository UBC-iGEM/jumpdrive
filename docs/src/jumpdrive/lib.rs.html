14/// The internal representation of a Jumpdrive process.\
15/// **Note**: this type should never be constructed directly.
16/// Instead, use the `jumpdrive!` macro.
17#[derive(Debug)]
18pub struct Jumpdrive {
19    map: Map<&'static str, (&'static [u8], &'static str)>,
20    socket: Option<Socket>,
21    other_paths: Map<&'static str, fn(&mut TcpStream)>,
22    error_handler: fn(String),
23}
24type Socket = (&'static str, fn(&mut WebSocket<TcpStream>));
25
26/// A specialized Result type for the `jumpdrive!` macro.
27/// # Returns
28/// - Err if binding to a TcpListener fails on the requested IP and PORT
29/// - Should never realistically return otherwise
30pub type JumpdriveResult = io::Result<()>;
31
32impl Jumpdrive {
33    pub fn new(
34        map: Map<&'static str, (&'static [u8], &'static str)>,
35        socket: Option<Socket>,
36        other_paths: Map<&'static str, fn(&mut TcpStream)>,
37        error_handler: fn(String),
38    ) -> Self {
39        Self {
40            map,
41            socket,
42            other_paths,
43            error_handler,
44        }
45    }
46
47    pub fn serve(self) -> JumpdriveResult {
48        let config = Arc::new(self);
49        let ip = env::var("IP").unwrap_or("127.0.0.1".to_string());
50        let port = env::var("PORT").unwrap_or("9999".to_string());
51        let addr = format!("{ip}:{port}");
52        let listener = TcpListener::bind(addr)?;
53
54        for connection in listener.incoming() {
55            if let Ok(conn) = connection
56                && let Ok(mut conn_clone) = conn.try_clone()
57            {
58                let config = Arc::clone(&config);
59                thread::spawn(move || {
60                    let output = || -> Result<(), Box<dyn Error>> {
61                        let mut buffer = [0; 1024];
62                        let len = conn_clone.peek(&mut buffer)?;
63                        if len == 0 {
64                            return Err("Request is empty!".into());
65                        }
66                        let request = String::from_utf8_lossy(&buffer[0..len]);
67                        if let Some(line) = request.lines().next() {
68                            let components: Vec<_> = line.split_whitespace().collect();
69                            if let [method, mut path, protocol] = components[0..3]
70                                && method == "GET"
71                                && protocol == "HTTP/1.1"
72                            {
73                                {
74                                    if path == "/" {
75                                        path = "/index.html";
76                                    }
77                                    if let Some((socket_path, socket_handler)) = config.socket
78                                        && path == socket_path
79                                    {
80                                        let mut ws = accept(conn_clone)?;
81                                        socket_handler(&mut ws);
82                                        return Ok(());
83                                    }
84                                    if let Some((asset, content_type)) = config.map.get(&path[1..])
85                                    {
86                                        let response = generate_response(content_type, asset);
87                                        conn_clone.write_all(&response)?;
88                                        return Ok(());
89                                    }
90
91                                    if let Some(callback) = config.other_paths.get(path) {
92                                        callback(&mut conn_clone);
93                                    }
94                                }
95                            } else {
96                                return Err("Malformed HTTP request header!".into());
97                            }
98                        }
99                        Ok(())
100                    };
101                    if let Err(e) = output() {
102                        (config.error_handler)(e.to_string());
103                    }
104                });
105            } else {
106                (config.error_handler)("Failed to connect to a client!".to_string());
107            }
108        }
109        Ok(())
110    }
111}

113fn generate_response(ty: &'static str, content: &'static [u8]) -> Vec<u8> {
114    let len = content.len();
115    let mut response =
116        format!("HTTP/1.1 200 OK\r\nContent-Length: {len}\r\nContent-Type: {ty}\r\n\r\n")
117            .into_bytes();
118    response.extend_from_slice(content);
119    response
120}
